131 constexpr auto vdc(unsigned long count, unsigned long base) -> double {
132 auto reslt = 0.0;
133 auto denom = 1.0;
134 while (count != 0) {
135 const auto remainder = count % base;
136 count /= base;
137 denom *= double(base);
138 reslt += double(remainder) / denom;
139 }
140 return reslt;
141 }
